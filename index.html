function renderProfilePills(){ const {profiles, activeId} = getProfilesAndActive(); const frag = document.createDocumentFragment(); profiles.forEach(p=>{ const b=document.createElement('button'); b.className='profile-pill'+(p.id===activeId?' active':''); b.dataset.id=p.id; b.textContent=p.name; frag.appendChild(b); }); const add=document.createElement('button'); add.className='profile-pill add'; add.dataset.action='add-profile'; add.textContent='+ Add profile'; frag.appendChild(add); // NEW: settings location const settings = document.getElementById('profilePillsSettings'); if(settings){ settings.innerHTML=''; settings.appendChild(frag.cloneNode(true)); } // (If you also want them on Profile screen header, you can append there too) }
